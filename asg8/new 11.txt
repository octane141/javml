<script> 
function validateEmail(emailId)
{
var mailformat = /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/;
if(emailId.value.match(mailformat))
{
    alert("Email Address Valid!")
document.form1.text1.focus();
return true;
}
else
{
alert("Invalid Email Address!");
document.form1.text1.focus();
return false;
}
}    
</script>
